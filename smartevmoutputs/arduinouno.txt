
#include <LiquidCrystal.h>

LiquidCrystal lcd(2, 3, 4, 5, 6, 7); // Initialize LCD with Arduino pins

const int buzzer_Pin = 8; // Pin for the buzzer
const int button_1 = 9; // Button 1 for Party 1
const int button_2 = 10; // Button 2 for Party 2
const int button_3 = 11; // Button 3 for Party 3

bool button_1_status = LOW; // Status for button 1
bool button_2_status = LOW; // Status for button 2
bool button_3_status = LOW; // Status for button 3

static int party_1_count = 0; // Vote count for Party 1
static int party_2_count = 0; // Vote count for Party 2
static int party_3_count = 0; // Vote count for Party 3

String winner_name = ""; // Variable to store winner name
int total_votes_cast = 0; // Track total number of votes

void setup() {

  pinMode(buzzer_Pin, OUTPUT);

  pinMode(button_1, INPUT);
  pinMode(button_2, INPUT);
  pinMode(button_3, INPUT);

  Serial.begin(9600); // Initialize serial communication

  lcd.begin(16, 2); // Initialize LCD

  lcd.print("Electronic");
  lcd.setCursor(3, 1);
  lcd.print("Vote Machine");

  delay(2000);
}

void loop() {

  lcd.clear();
  lcd.print("Scan Your Face");

  delay(2000);

  // Simulate face recognition approval
  bool isFaceRecognized = simulateFaceRecognition();

  if (isFaceRecognized) {

    lcd.clear();
    lcd.print("Face Recognized");

    delay(1000);

    lcd.clear();
    lcd.print("Cast Your Vote");

    delay(1000);

    // Wait for user to press a button
    button_1_status = digitalRead(button_1);
    button_2_status = digitalRead(button_2);
    button_3_status = digitalRead(button_3);

    if (button_1_status == HIGH) {

      party_1_count++;
      total_votes_cast++;

      lcd.clear();
      lcd.print("You Voted:");
      lcd.setCursor(0, 1);
      lcd.print("Party 1");

      Serial.println("Vote cast for Party 1");

      buzzAndDelay();

    } 
    else if (button_2_status == HIGH) {

      party_2_count++;
      total_votes_cast++;

      lcd.clear();
      lcd.print("You Voted:");
      lcd.setCursor(0, 1);
      lcd.print("Party 2");

      Serial.println("Vote cast for Party 2");

      buzzAndDelay();

    } 
    else if (button_3_status == HIGH) {

      party_3_count++;
      total_votes_cast++;

      lcd.clear();
      lcd.print("You Voted:");
      lcd.setCursor(0, 1);
      lcd.print("Party 3");

      Serial.println("Vote cast for Party 3");

      buzzAndDelay();

    } 
    else {

      lcd.clear();
      lcd.print("No Vote Cast");

      delay(2000);
    }

    // Check if voting is complete
    if (total_votes_cast >= 3) { // Example: End voting after 3 votes

      determineWinner();

      lcd.clear();
      lcd.print("Winner is:");
      lcd.setCursor(0, 1);
      lcd.print(winner_name);

      delay(5000);

      resetVoting();
    }

  } 
  else {

    lcd.clear();
    lcd.print("Face Not");
    lcd.setCursor(0, 1);
    lcd.print("Recognized");

    delay(2000);
  }
}

bool simulateFaceRecognition() {

  // Simulate the result of face recognition
  return true; // Replace with actual integration in future
}

// Buzzer Function
void buzzAndDelay() {

  digitalWrite(buzzer_Pin, HIGH);
  delay(500);

  digitalWrite(buzzer_Pin, LOW);
  delay(500);
}

// Determine Winner Function
void determineWinner() {

  if (party_1_count > party_2_count && party_1_count > party_3_count) {

    winner_name = "Party 1";

  } 
  else if (party_2_count > party_1_count && party_2_count > party_3_count) {

    winner_name = "Party 2";

  } 
  else if (party_3_count > party_1_count && party_3_count > party_2_count) {

    winner_name = "Party 3";

  } 
  else {

    winner_name = "Tie";
  }

  Serial.println("Voting Complete.");

  Serial.print("Party 1:");
  Serial.println(party_1_count);

  Serial.print("Party 2:");
  Serial.println(party_2_count);

  Serial.print("Party 3:");
  Serial.println(party_3_count);

  Serial.print("Winner:");
  Serial.println(winner_name);
}

// Reset Voting Function
void resetVoting() {

  party_1_count = 0;
  party_2_count = 0;
  party_3_count = 0;

  total_votes_cast = 0;

  winner_name = "";

  lcd.clear();
  lcd.print("Vote Machine");

  delay(2000);
}
